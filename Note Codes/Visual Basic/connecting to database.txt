        Dim command As MySqlCommand
        Dim reader As MySqlDataReader

        connect.ConnectionString = "server=localhost;user id=root;
				password=Jedidiah1;persistsecurityinfo=True;database=accounts"
        
	Try
            connect.Open()

            MessageBox.Show("Succesfully connected!")
            Dim Query As String
            Query = "SELECT * FROM accounts.user WHERE username='Nostee' and password='Jedidiah1'"
            command = New MySqlCommand(Query, connect)
