class User
{	
	String username; 	
	int age;
	
	User(String a, int b) 	
	{
		this.username = a;
		this.age = b;
	}

	void login() 
	{
		print("User logged in.");	
	}
}

class SuperUser extends User
{
  SuperUser(String superA, int superB) : super(superA,superB);

	void publish()
	{
		print("Testing only.");
	}
}
